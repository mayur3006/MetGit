import BankPolicy.*;

public class SITest {
	private static double SI(double p,int n,IPolicy P){
		double r=P.getRate();
		double Si=(p*n*r)/100;
		return Si;
	}
	
	public static void main(String[] args) {
		double amount;
		int period;
		amount=Double.parseDouble(args[0]);
		period=Integer.parseInt(args[1]);
		
		BankA B1=new BankA();
		BankB B2=new BankB();
		BankC B3=new BankC();
		
		System.out.printf("Simple Interest for BankA:%.2f%n",SI(amount,period,B1));
		System.out.printf("Simple Interest for BankB:%.2f%n",SI(amount,period,B2));
		System.out.printf("Simple Interest for BankC:%.2f%n",SI(amount,period,B3));
	}
}
